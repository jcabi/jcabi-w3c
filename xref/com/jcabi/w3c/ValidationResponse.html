30  package com.jcabi.w3c;
31  
32  import java.net.URI;
33  import java.nio.charset.Charset;
34  import java.util.Set;
35  
36  /**
37   * Response of HTML or CSS validation.
38   *
39   * <p>See {@link ValidatorBuilder} for explanation of how to get an instance
40   * of this interface.
41   *
42   * <p>Implementation may be mutable but thread-safe.
43   *
44   * @author Yegor Bugayenko (yegor@tpc2.com)
45   * @version $Id: c976525cb433993d9701f9766a095991b45aefbc $
46   * @see ValidatorBuilder
47   * @see Validator#validate(String)
48   * @see <a href="http://validator.w3.org/docs/api.html">W3C API, HTML</a>
49   * @see <a href="http://jigsaw.w3.org/css-validator/api.html">W3C API, CSS</a>
50   * @since 0.1
51   */
52  public interface ValidationResponse {
53  
54      /**
55       * The document is valid and has no errors or warnings?
56       * @return Is it valid?
57       */
58      boolean valid();
59  
60      /**
61       * Who checked the document (normally contains a URL of W3C server).
62       * @return URI of the server
63       */
64      URI checkedBy();
65  
66      /**
67       * DOCTYPE of the document, if detected by the validator (may be empty
68       * if {@code DOCTYPE} is not detected or if it's a CSS document).
69       * @return Doctype or empty string
70       */
71      String doctype();
72  
73      /**
74       * Charset of the document, if detected by the server (may be empty
75       * if charset is not detected or it's a CSS document).
76       * @return Charset of the document, e.g. {@code "UTF-8"}
77       */
78      Charset charset();
79  
80      /**
81       * Returns list of errors found during validation.
82       * @return List of errors or an empty list if no errors found
83       */
84      Set<Defect> errors();
85  
86      /**
87       * Returns lsit of warnings found during validation.
88       * @return List of warnings
89       */
90      Set<Defect> warnings();
91  
92  }
